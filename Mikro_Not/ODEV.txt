#INCLUDE <P16F877A.INC>
SAYAC0 EQU 0X20
SAYAC1 EQU 0X21

	ORG 0X00
	GOTO MAIN
	ORG 0X04
KESME
	BCF PIR1,ADIF
	BANKSEL ADRESH
	MOVFW ADRESH
	BANKSEL PORTB
	MOVWF PORTB
	CALL GECIKME
	BANKSEL ADCON0
	BSF ADCON0,GO
	RETFIE
MAIN
	BANKSEL TRISA
	BSF TRISA,0
	CLRF TRISB
	BANKSEL PORTA
	CLRF PORTA
	CLRF PORTB
	
	BANKSEL ADCON1
	CLRF ADCON1
	BANKSEL ADCON0
	MOVLW B'11000001'
	MOVWF ADCON0
	CALL GECIKME
	BANKSEL PIR1
	BCF PIR1,ADIF
	BANKSEL PIE1
	BSF PIE1,ADIE
	BANKSEL INTCON
	BSF INTCON,GIE
	BSF INTCON,PEIE
	BANKSEL ADCON0
	BSF ADCON0,GO
DONGU
	GOTO DONGU
	
GECIKME
	MOVLW D'255'
	MOVWF SAYAC0
GECIKME1
	MOVLW D'255'
	MOVWF SAYAC1
GECIKTIR
	DECFSZ SAYAC1,F
	GOTO GECIKTIR
	DECFSZ SAYAC0,F
	GOTO GECIKME1
	RETURN
	
	END
